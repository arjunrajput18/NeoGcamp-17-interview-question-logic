var txt = document.querySelector("#txt-input");
var btnSubmit = document.querySelector("#btn-submit");
var outputDIv = document.querySelector("#output");

function checkHandler() {
  btnSubmit.disabled = true;
  txtVal = txt.value;
  console.log(txtVal);
  if (txtVal == "" || txtVal == null) {
    btnSubmit.disabled = true;
    outputDIv.innerText = "Enter password";
  } else if (txtVal.length >= 10) {
    txt.style.backgroundColor = "green";
    outputDIv.innerText = "success";
    btnSubmit.disabled = false;
  } else if (txtVal.length < 10) {
    txt.style.backgroundColor = "red";
    outputDIv.innerHTML = "password is lesser than 10";
    btnSubmit.disabled = true;
  }
}
