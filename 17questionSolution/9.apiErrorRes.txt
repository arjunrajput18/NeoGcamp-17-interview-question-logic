let serverUrl = 'https://jsonplaceholder.typicode.com/users/';

function apiHandler() {
  fetch(serverUrl)
   .then(res => res.json())
  .then(data => output.innerText = data.error)
  .catch(error => output.innerText = error)
  
}
btnApi.addEventListener("click", apiHandler);
